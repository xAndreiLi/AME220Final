    a3 = a * 16^1 + 3 * 16^0 = 10 * 16 + 3 * 1 = 163

    b6 = b * 16^1 + 6 * 16^0 = 11 * 16 + 6 * 1 = 182

    #37feb2 = rgb(55, 254, 178)
    37 = 3 * 16^1 + 7 * 16^0 = 3 * 16 + 7 * 1 = 55
